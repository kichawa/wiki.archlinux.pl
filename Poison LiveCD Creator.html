[edytuj] Wprowadzenie

Poison LiveCD Creator jest narzędziem przeznaczonym do prostego utworzenia własnego LiveCD z Archem, do czego wykorzystuje Makefile. Oczywiście pozwala na dodanie własnych repozytoriów oraz pakietów oraz skonfigurowania nowopowstałego LiveCD do swoich potrzeb.

Pakiet znajduje się w Arch User Repository.

[edytuj] Readme

Notka: W quick steps opisane są kroki, jakie trzeba kolejno wykonać. Opisane są one bardziej szczegółowo w detailed steps, good one to follow.
File: README
----------------------- poison live cd creator -----------------

quick steps
{
  root@poison# make mkdir               			//to make the directory structures of your live cd
  root@poison# make copy-pac            			//copies necessary files to your live cd dir.
  root@poison# make pacman-sync         		        //create package DB for your live cd
  root@poison# make mount               			//mount the VFS to ur live cd root dir.
  root@poison# make pacman-base         		        //installs base packages
  root@poison# make pacman-squashfs		                //install squasfs modules needed for live cd
  root@poison# make pacman-aufs			                // installs aufs module
  root@poison# make pacman-kernel26     		        //reinstall kernel26 and initramfs
  root@poison# make mkinitramfs         		        //create
  root@poison# make mksquashfs          		        // compress and make squashfs file of your live cd root dir
  root@poison# make iso-image           			// your live cd is ready
}

detailed steps, good one to follow
{
  root@poison# make mkdir               			//to make the directory structures of your live cd
						                //      root - root directory of your live cd
								//      boot - contains boot files for booting live cd
								//             not same as root/boot
								//      iso  - contains the files in fianl live cd

  root@poison# make copy-pac            			//copies necessary files to your live cd dir.
								// it copies the file ./conf/pacman.conf to ./root/etc/pacman.con

  root@poison# make pacman-sync         		        //create package DB for your live cd
								//you need net connection for this.
								//it uses the configuration file from ./conf/pacman.conf
								//you can edit this file for your need.
								//this uses /etc/pacman.d/mirrorlist for mirrors, not the mirrorlist in ./root/etc/pacman.d/mirrorlist. so set it correct.

  root@poison# make mount               			//mount the VFS to ur live cd root dir.

  root@poison# make pacman-base         		        //installs base packages

  root@poison# make pacman-squashfs     		        // from here on pacman uses the conf file ./root/etc/pacman.conf not ./conf/pacman.conf
							 	// make your changes respectively.
								// if u r using 64 bit linux, you may have to change "Architecture = auto" to "Architecture = i686" in
								// ./root/etc/pacman.conf file, if you want 32 bit live cd.

  root@poison# make pacman-aufs         		        // installs aufs module for live cd.

  root@poison# make pacman-kernel26     		        //reinstall kernel26 and initramfs
								// mkinitcpio is used to create initramfs.
								// so check out ./root/etc/mkinitcpio.conf file and make necessary changes if any.
								// this file should contain MODULES="aufs squashfs loop" which are used by init script
								// and remove autodetect from hooks and add usb.
								// the above 2 changes are made by makefile automatically, its advised to check it again.

 root@poison# make mkinitramfs         			        //copies and creates initramfs for live cd.
								// this uses the init script in ./conf/init
								// its just an shell script. you should be able to understand it.
								// try it out. its of more fun and interest. :D

  root@poison# make pacman-adons
  root@poison# make pacman-xorg
  root@poison# make pacman-xfce
								// add package which u want to be in live cd to ADONS variable in Makefile
								// the above 3 commands installs xorg, xfce4.
								// manually edit ./root/etc/rc.conf to add start up service
								// it possible create your own script in ./root/etc/rc.d/ to run xfce-session automatically.
					                        // by default root password is not set and needed to login. you can set one by using "chroot ./root passwd"
					                        // you can make all changes like editing thi ./boot/grub/menu.lst to change the menu entry name
					                        // edit ./root/etc/rc.conf to set hostname.
					                        // edit ./root/etc/rc.sysinit to display your name instead of "Arch linu" while booting etc......
  
root@poison# make mksquashfs       			        // compress and make squashfs file of your live cd root dir
										// the compressed file is placed in ./iso as file root.squashfs

  root@poison# make iso-image           			// your live cd is ready

}

all:
	echo 'please read the document for making the live cd..... its README file in current dir :P' 
	echo 'thank you :D' 

copy-pac:
	cp ./conf/pacman.conf ./root/etc/
	cp ./conf/mirrorlist ./root/etc/pacman.d/
mount:
	mount -t sysfs sysfs "root/sys"
	mount -t proc proc "root/proc"
	mount -o bind /dev "root/dev"
	mount -t tmpfs shm "root/dev/shm"
	mount -t devpts devpts "root/dev/pts"
	mount --bind /var/run "root/var/run"
	mount --bind /var/lib/dbus "root/var/lib/dbus"
	mount --bind /var/cache/pacman/pkg root/var/cache/pacman/pkg
umount:
	umount "root/sys" ; \
	umount "root/proc" ; \
	umount "root/dev/shm" ; \
	umount "root/dev/pts" ; \
	umount "root/dev" ; \
	umount "root/var/run" ; \
	umount "root/var/cache/pacman/pkg" ; \
	umount "root/var/lib/dbus" ; echo "----------------umount done-------------- "

mkdir: 
	mkdir -p boot root/{var/{lib/{pacman,dbus},cache/pacman/pkg,run},dev,proc,sys,etc/pacman.d,root,tmp} squashmod mkinitrd iso
clean: umount
	rm -rf boot squashmod mkinitrd root

pacman-base:
	rm root/etc/pacman.conf root/etc/pacman.d/mirrorlist;\
	pacman --config ./conf/pacman.conf --root ./root -S base
pacman-sync:
	pacman --config ./conf/pacman.conf --root ./root -Sy
pacman-kernel26:
	sed -i s/autodetect/'usb lvm2'/ ./root/etc/mkinitcpio.conf;\
	sed -i s/'MODULES=""'/'MODULES="aufs squashfs loop"'/ ./root/etc/mkinitcpio.conf;\
	pacman --config ./root/etc/pacman.conf --root ./root -S kernel26
pacman-squashfs:
	pacman --config ./root/etc/pacman.conf --root ./root -S squashfs-tools
pacman-aufs:
	pacman --config ./root/etc/pacman.conf --root ./root -S aufs2 #aufs2-util
pacman-update:
	pacman --config ./root/etc/pacman.conf --root ./root -Su
pacman-adons:
	pacman --root ./root/ --config ./root/etc/pacman.conf -S $(ADONS)

mkinitramfs:
	cp -rf ./root/boot/* ./boot
	cp ./root/usr/lib/grub/i386-pc/stage2_eltorito ./boot/grub/
	rm ./boot/kernel26-fallback.img
	mkdir ./boot/ramfs -p
	gzip -d ./boot/kernel26.img -c > ./boot/ramfs/_cpio
	rm ./boot/kernel26.img
	cd ./boot/ramfs;bsdtar -x -f _cpio; rm _cpio;
	rm ./boot/ramfs/init
	cp ./conf/init ./boot/ramfs/
	cd ./boot/ramfs; find .

W ADONS, XORG i XFCE są zawarte pliki do instalacji. Zainstalujesz je na LiveCD wydając odpowiednio polecenie:

$ make pacman-adons
$ make pacman-xorg
$ make pacman-xfce

Oczywiście możesz sobie zmienić pakiety, na przykład zamiast XFCE możesz wybrać GNOME.
